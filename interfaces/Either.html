Interface Either<L, R>

The Either interface, representing a value that is either Left (usually an error or alternate value) or Right (typically a success value). It is a powerful way to handle operations that can result in two distinctly different types of outcomes.

interface Either<L, R> {
    type: symbol;
    isLeft(): this is LeftEither<L, R>;
    isRight(): this is RightEither<L, R>;
    left(): Option<L>;
    leftAndThen<U>(fn: ((val: L) => Either<U, R>)): Either<U, R>;
    mapLeft<U>(fn: ((val: L) => U)): Either<U, R>;
    mapRight<U>(fn: ((val: R) => U)): Either<L, U>;
    match<U>(fn: Match<L, R, U>): U;
    right(): Option<R>;
    rightAndThen<U>(fn: ((val: R) => Either<L, U>)): Either<L, U>;
    unwrap(): L | R;
    unwrapLeft(): L;
    unwrapLeftOr(other: L): L;
    unwrapRight(): R;
    unwrapRightOr(other: R): R;
}
